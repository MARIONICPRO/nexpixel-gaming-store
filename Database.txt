-- 1. TABLA USUARIOS
CREATE TABLE usuarios (
    id_usuario SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255),
    tipo_usuario VARCHAR(20) CHECK (tipo_usuario IN ('cliente', 'proveedor', 'admin')) DEFAULT 'cliente',
    foto_perfil TEXT,
    empresa VARCHAR(100),
    nit VARCHAR(20),
    telefono VARCHAR(20),
    descripcion TEXT,
    verificado BOOLEAN DEFAULT FALSE,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ultima_conexion TIMESTAMP
);

-- 2. TABLA CATEGORIA
CREATE TABLE categoria (
    id_categoria SERIAL PRIMARY KEY,
    nombre_grupo VARCHAR(50),
    grupo VARCHAR(50)
);

-- 4. TABLA PRODUCTO
CREATE TABLE producto (
    id_producto SERIAL PRIMARY KEY,
    id_categoria INT REFERENCES categoria(id_categoria),
    id_plataforma INT REFERENCES plataforma(id_plataforma),
    id_proveedor INT REFERENCES usuarios(id_usuario),
    nombre_producto VARCHAR(200) NOT NULL,
    descripcion TEXT,
    descripcion_corta VARCHAR(255),
    precio DECIMAL(10,2) NOT NULL,
    precio_original DECIMAL(10,2),
    descuento INT DEFAULT 0,
    tipo_producto VARCHAR(50) CHECK (tipo_producto IN ('Juego', 'Tarjeta regalo', 'Suscripcion', 'Consola', 'Accesorio')),
    stock INT DEFAULT 0,
    imagen_url TEXT,
    imagenes_adicionales TEXT[],
    plataforma_juego VARCHAR(50),
    genero VARCHAR(50),
    edicion VARCHAR(50),
    fecha_lanzamiento DATE,
    desarrollador VARCHAR(100),
    valor_tarjeta DECIMAL(10,2),
    moneda VARCHAR(10) DEFAULT 'COP',
    visitas INT DEFAULT 0,
    ventas_totales INT DEFAULT 0,
    estado VARCHAR(20) DEFAULT 'activo',
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5. TABLA INVENTARIO_LLAVES (Códigos digitales)
CREATE TABLE inventario_llaves (
    id_llave SERIAL PRIMARY KEY,
    id_producto INT REFERENCES producto(id_producto),
    codigo TEXT NOT NULL,
    estado VARCHAR(20) CHECK (estado IN ('disponible', 'vendida', 'reservada', 'expirada')) DEFAULT 'disponible',
    fecha_vencimiento DATE,
    id_detalle INT,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 6. TABLA CARRITO
CREATE TABLE carrito (
    id SERIAL PRIMARY KEY,
    id_cuenta INT REFERENCES usuarios(id_usuario) UNIQUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado VARCHAR(20) CHECK (estado IN ('activo', 'abandonado', 'convertido')) DEFAULT 'activo'
);

-- 7. TABLA CARRITO_ITEMS
CREATE TABLE carrito_items (
    id SERIAL PRIMARY KEY,
    carrito_id INT REFERENCES carrito(id) ON DELETE CASCADE,
    id_producto INT REFERENCES producto(id_producto),
    cantidad INT DEFAULT 1,
    precio_unitario DECIMAL(10,2),
    fecha_agregado TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 8. TABLA COMPRA
CREATE TABLE compra (
    id_compra SERIAL PRIMARY KEY,
    id_cuenta INT REFERENCES usuarios(id_usuario),
    fecha_compra TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total DECIMAL(10,2) NOT NULL,
    subtotal DECIMAL(10,2),
    impuestos DECIMAL(10,2) DEFAULT 0,
    descuento_total DECIMAL(10,2) DEFAULT 0,
    estado VARCHAR(30) CHECK (estado IN ('pendiente', 'pagado', 'enviado', 'entregado', 'cancelado', 'reembolsado')) DEFAULT 'pendiente',
    metodo_pago VARCHAR(50),
    comprobante_pago TEXT,
    notas TEXT
);

-- 9. TABLA DETALLE_COMPRA
CREATE TABLE detalle_compra (
    id_detail SERIAL PRIMARY KEY,
    id_compra INT REFERENCES compra(id_compra) ON DELETE CASCADE,
    id_producto INT REFERENCES producto(id_producto),
    cantidad INT DEFAULT 1,
    precio_unitario DECIMAL(10,2) NOT NULL,
    descuento_aplicado DECIMAL(10,2) DEFAULT 0,
    subtotal DECIMAL(10,2) NOT NULL,
    id_llave INT REFERENCES inventario_llaves(id_llave)
);

-- 10. TABLA PAGO
CREATE TABLE pago (
    id_pago SERIAL PRIMARY KEY,
    id_compra INT REFERENCES compra(id_compra),
    id_metodo VARCHAR(50),
    fecha_pago TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado_pago VARCHAR(20) CHECK (estado_pago IN ('pendiente', 'completado', 'rechazado')) DEFAULT 'pendiente',
    monto DECIMAL(10,2),
    transaccion_id VARCHAR(100)
);

-- 11. TABLA DEVOLUCION
CREATE TABLE devolucion (
    id_devolucion SERIAL PRIMARY KEY,
    id_compra INT REFERENCES compra(id_compra),
    id_usuario INT REFERENCES usuarios(id_usuario),
    motivo TEXT,
    fecha_solicitud TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado VARCHAR(20) CHECK (estado IN ('pendiente', 'aprobada', 'rechazada')) DEFAULT 'pendiente',
    monto DECIMAL(10,2),
    fecha_resolucion TIMESTAMP
);

-- 12. TABLA HISTORIAL_VENTAS
CREATE TABLE historial_ventas (
    id_historial SERIAL PRIMARY KEY,
    id_cuenta INT REFERENCES usuarios(id_usuario),
    id_compra INT REFERENCES compra(id_compra),
    fecha_servicio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    leido BOOLEAN DEFAULT FALSE
);

-- 13. TABLA RECOMENDACION (VERSIÓN SIMPLE Y FUNCIONAL)
CREATE TABLE recomendacion (
    id_recomendacion SERIAL PRIMARY KEY,
    id_usuario INT REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    id_producto INT REFERENCES producto(id_producto) ON DELETE CASCADE,
    tipo_recomendacion VARCHAR(30) CHECK (tipo_recomendacion IN ('personalizada', 'popular', 'similar', 'complementario')),
    puntuacion FLOAT,
    fecha_generacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    seccion VARCHAR(50),
    visto BOOLEAN DEFAULT FALSE,
    clickeado BOOLEAN DEFAULT FALSE,
    fecha_visto TIMESTAMP,
    fecha_click TIMESTAMP
);

-- 14. TABLA INTERACCIONES_USUARIO
CREATE TABLE interacciones_usuario (
    id_interaccion SERIAL PRIMARY KEY,
    id_usuario INT REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    id_producto INT REFERENCES producto(id_producto) ON DELETE CASCADE,
    tipo_interaccion VARCHAR(20) CHECK (tipo_interaccion IN ('vista', 'carrito', 'compra', 'click_recomendacion', 'busqueda')),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    valor INT DEFAULT 1,
    sesion_id VARCHAR(100)
);

-- 15. TABLA PREFERENCIAS_USUARIO
CREATE TABLE preferencias_usuario (
    id_preferencia SERIAL PRIMARY KEY,
    id_usuario INT REFERENCES usuarios(id_usuario) ON DELETE CASCADE UNIQUE,
    plataforma_favorita INT REFERENCES plataforma(id_plataforma),
    categoria_favorita INT REFERENCES categoria(id_categoria),
    rango_precio_favorito VARCHAR(20) CHECK (rango_precio_favorito IN ('bajo', 'medio', 'alto')),
    datos_json JSONB,
    ultima_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 16. TABLA BUSQUEDAS_USUARIO
CREATE TABLE busquedas_usuario (
    id_busqueda SERIAL PRIMARY KEY,
    id_usuario INT REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    termino_busqueda TEXT NOT NULL,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    resultados_encontrados INT,
    hizo_click BOOLEAN DEFAULT FALSE,
    producto_clickeado_id INT REFERENCES producto(id_producto)
);

-- 17. TABLA METRICAS_PRODUCTOS
CREATE TABLE metricas_productos (
    producto_id INT REFERENCES producto(id_producto) ON DELETE CASCADE PRIMARY KEY,
    veces_visto INT DEFAULT 0,
    veces_anadido_carrito INT DEFAULT 0,
    veces_comprado INT DEFAULT 0,
    conversion_rate FLOAT DEFAULT 0,
    popularidad_general FLOAT DEFAULT 0,
    tendencia VARCHAR(20) CHECK (tendencia IN ('subiendo', 'estable', 'bajando')) DEFAULT 'estable',
    ultima_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- ÍNDICES PARA MEJORAR RENDIMIENTO
-- ============================================

CREATE INDEX idx_productos_nombre ON producto(nombre_producto);
CREATE INDEX idx_productos_categoria ON producto(id_categoria);
CREATE INDEX idx_productos_plataforma ON producto(id_plataforma);
CREATE INDEX idx_productos_precio ON producto(precio);
CREATE INDEX idx_productos_tipo ON producto(tipo_producto);
CREATE INDEX idx_productos_proveedor ON producto(id_proveedor);
CREATE INDEX idx_interacciones_usuario ON interacciones_usuario(id_usuario, fecha);
CREATE INDEX idx_interacciones_producto ON interacciones_usuario(id_producto, fecha);
CREATE INDEX idx_recomendaciones_usuario ON recomendacion(id_usuario, fecha_generacion);
CREATE INDEX idx_recomendaciones_no_vistas ON recomendacion(id_usuario, visto) WHERE visto = false;
CREATE INDEX idx_compras_usuario ON compra(id_cuenta, fecha_compra);
CREATE INDEX idx_compras_estado ON compra(estado);
CREATE INDEX idx_inventario_producto ON inventario_llaves(id_producto, estado) WHERE estado = 'disponible';
CREATE INDEX idx_inventario_estado ON inventario_llaves(estado);
CREATE INDEX idx_historial_usuario ON historial_ventas(id_cuenta, fecha_servicio);

-- ============================================
-- FUNCIONES Y TRIGGERS
-- ============================================

CREATE OR REPLACE FUNCTION actualizar_metricas_producto()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO metricas_productos (producto_id, veces_visto, veces_comprado, ultima_actualizacion)
    VALUES (NEW.id_producto, 1, 0, NOW())
    ON CONFLICT (producto_id) 
    DO UPDATE SET 
        veces_visto = metricas_productos.veces_visto + 1,
        ultima_actualizacion = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_actualizar_metricas
AFTER INSERT ON interacciones_usuario
FOR EACH ROW
WHEN (NEW.tipo_interaccion = 'vista')
EXECUTE FUNCTION actualizar_metricas_producto();

CREATE OR REPLACE FUNCTION actualizar_fecha_producto()
RETURNS TRIGGER AS $$
BEGIN
    NEW.fecha_actualizacion = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_actualizar_producto
BEFORE UPDATE ON producto
FOR EACH ROW
EXECUTE FUNCTION actualizar_fecha_producto();

CREATE OR REPLACE FUNCTION actualizar_fecha_carrito()
RETURNS TRIGGER AS $$
BEGIN
    NEW.fecha_actualizacion = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_actualizar_carrito
BEFORE UPDATE ON carrito
FOR EACH ROW
EXECUTE FUNCTION actualizar_fecha_carrito();
